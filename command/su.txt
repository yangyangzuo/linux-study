documents:



description:
  切换用户，并且切换目录
  exit退出当前用户，返回原始用户状态

help:
    su --h
    su --help
    man su
    

example:
    #带上-可以切换到用户的家目录，如果不带-,则保持当前的路径不变

    #切换到对应的用户,并且目录为这个用户的家目录
    su - 用户名
    
    #仅仅切换到对应的用户,当前的目录路径不变
    su 用户名

    #如果省略用户名,则默认是root账户
    su
    #等价于
    su root

    #默认情况下,root账户在每次开机的时候密码都会更改,需要对root用户设置密码后,才可以切换到root用户
    sudo passwd root

    #如果要退出当前用户，使用exit命令
    #如果由用户a -> 切换到用户b -> 切换到用户c
    #由用户c -> 执行exit会切换到用户b -> 再执行exit,则切换到用户a -> 再执行exit，则退出终端
